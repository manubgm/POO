import java.awt.*;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import javax.imageio.ImageIO;

public class Mario {
    private int x, y;
    private int largura, altura;
    private int escala = 4; // sprite original é 16x16, fica 64x64

    // Sprites recortados do spritesheet (224x32 = 14 sprites de 16x16 por linha)
    // Sugestão de índices (confira no spritesheet):
    //   0       -> parado
    //   1, 2, 3 -> andando (animação cíclica)
    //   5       -> pulando
    private BufferedImage[] sprites;

    // Estado de desenho
    protected int frameAtual = 0;
    protected boolean olhandoDireita = true;

    //construtor
    public Mario(int x, int alturaTela) throws IOException {
        BufferedImage sheet = ImageIO.read(new File("../sprites/spritesheet.png"));
        sprites = new BufferedImage[14];
        for (int i = 0; i < 14; i++) {
            sprites[i] = sheet.getSubimage(i * 16, 0, 16, 16);
        }

        this.largura = 16 * escala;
        this.altura  = 16 * escala;
        this.x = x;
        this.y = alturaTela - altura; // começa no chão
        this.chaoH = this.y;//agora sei onde ta o chao
    }

    /**
     * Desenha o sprite atual na tela, espelhado horizontalmente
     * caso o Mario esteja olhando para a esquerda.
     */
    public void desenhar(Graphics g) {
        BufferedImage sprite = sprites[frameAtual];
        if (olhandoDireita) {
            g.drawImage(sprite, x, y, largura, altura, null);
        } else {
            // Truque: largura negativa = espelhamento horizontal
            g.drawImage(sprite, x + largura, y, -largura, altura, null);
        }
    }

    // TODO: Criar os métodos necessários para:
    // - Mover o Mario para esquerda/direita
    // - Fazer o Mario pular (com gravidade, formando uma parábola)
    // - Atualizar a física do pulo a cada frame
    // - Atualizar o sprite atual (frameAtual) conforme o estado:
    //     * parado, andando (alternar entre 3 frames) ou pulando
    // - Atualizar 'olhandoDireita' conforme a direção do movimento

    // variaveis para a fisica
    private double velX =0;
    private double velY = 0;
    private double gravidade = 0.6;
    private double forcaPulo = -12.0;
    private double velocidadeMov=5.0;//de andar
    private boolean noChao = true; //controle
    private int chaoH; //altura do chao


    // Getters
    public int getX() { return x; }
    public int getY() { return y; }
    public int getLargura() { return largura; }
    public int getAltura() { return altura; }

    // Setters de posição (úteis para a lógica de movimento e física)
    public void setX(int x) { this.x = x; }
    public void setY(int y) { this.y = y; }

    //metodos para o movimento
    public void movLeft(){
        velX = -velocidadeMov;
        olhandoDireita = false;
    }
    public void movRight(){
        velX = velocidadeMov;
        olhandoDireita = true;
    }
    public void stop(){
        velX=0;
    }
    public void jump(){
        if(noChao){
            velY = forcaPulo;
            noChao = false;
        }
    }

    //metodo pra fisica afetar a posição do mario
    public void atualizaFisica(){
        //aplica a gravidade
        if(!noChao){
            velY += gravidade;
        }

        //atualiza a posição
        x += velX;
        y += velY;

        //verifica colisão com o chão
        if(y >= chaoH){
            y = chaoH;
            velY = 0;
            noChao = true;
        }

        //verifica pra ele nao sair da tela pela lateral
        if(x<0)x=0;
        if(x > 800 - largura) x = 800 - largura;
    }

}
